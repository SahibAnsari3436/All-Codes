This code of apply child theme css code



// Enqueue the child theme stylesheet (loads after Blocksy)
add_action( 'wp_enqueue_scripts', function () {
    wp_enqueue_style(
        'blocksy-child-style',
        get_stylesheet_uri(), 
        [],
        filemtime( get_stylesheet_directory() . '/style.css' ) 
    );
}, 100);
